    Final implementation in createSubscriptionRecord:
    ```typescript
    async function createSubscriptionRecord(userId: string, event: PurchaseEvent): Promise<void> {
      if (!supabaseAdmin || !event.plan_id) return;

      // Resolve plan_id: stripe_products.id -> stripe_price_id -> plans.id
      const { data: product } = await supabaseAdmin
        .from('stripe_products')
        .select('stripe_price_id')
        .eq('id', event.plan_id)
        .single();

      if (!product?.stripe_price_id) {
        console.warn('[onboarding] Cannot create subscription: no stripe_price_id for product', event.plan_id);
        return;
      }

      const { data: plan } = await supabaseAdmin
        .from('plans')
        .select('id')
        .eq('stripe_price_id', product.stripe_price_id)
        .single();

      if (!plan) {
        console.warn('[onboarding] Cannot create subscription: no matching plan for stripe_price_id', product.stripe_price_id);
        return;
      }

      await supabaseAdmin.from('subscriptions').upsert({
        user_id: userId,
        plan_id: plan.id,
        stripe_customer_id: event.stripe_customer_id ?? null,
        stripe_subscription_id: event.stripe_subscription_id ?? null,
        status: 'active',
        current_period_start: new Date().toISOString(),
      }, { onConflict: 'user_id' });

      console.log(`[onboarding] Subscription record created for user ${userId}, plan ${plan.id}`);
    }
    ```
