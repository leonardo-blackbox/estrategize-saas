export interface CheckoutSessionResponse {
  url: string;
}

export function createCheckoutSession(priceId: string) {
  return client
    .post('/api/stripe/checkout-session', { json: { price_id: priceId } })
    .json<CheckoutSessionResponse>();
}

**File 2: frontend/src/features/planos/hooks/useCheckout.ts**

```typescript
import { useMutation } from '@tanstack/react-query';
import { createCheckoutSession } from '../../../api/checkout.ts';

export function useCheckout() {
  return useMutation({
    mutationFn: (priceId: string) => createCheckoutSession(priceId),
    onSuccess: (data) => {
      if (data.url) {
        window.location.href = data.url;
      }
    },
  });
}
```

The hook returns `{ mutate, isPending, error }`. The aggregator will use `mutate(planId)` and pass `isPending`/`error` state down to PlanCard.
