export async function adminUploadCourseBanner(courseId: string, file: File): Promise<{ banner_url: string }> {
  const { useAuthStore } = await import('../stores/authStore.ts');
  const session = useAuthStore.getState().session;
  const formData = new FormData();
  formData.append('file', file);
  const API_URL = (import.meta.env.VITE_API_URL as string) || 'http://localhost:3001';
  const res = await fetch(`${API_URL}/api/admin/courses/${courseId}/banner`, {
    method: 'POST',
    headers: session?.access_token ? { Authorization: `Bearer ${session.access_token}` } : {},
    body: formData,
  });
  // ... error handling
}
